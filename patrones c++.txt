#include <iostream>

using namespace std;

void mostrarPatronX() {
    char matrizLEDs[8][8];  // Crear una matriz 8x8 de caracteres

    // Inicializar la matriz con espacios en blanco
    for (int i = 0; i < 8; i++) {
        for (int j = 0; j < 8; j++) {
            matrizLEDs[i][j] = ' ';
        }
    }

    // Colocar 'X' en el patrÃ³n
    for (int i = 0; i < 8; i++) {
      matrizLEDs[i][i] = 'X';  // Diagonal de la esquina superior izquierda a la inferior derecha
      matrizLEDs[i][7 - i] = 'X';  // Diagonal de la esquina superior derecha a la inferior izquierda
    }

    for (int i = 0; i < 8; i++) {
        for(int j=0; i<8; i++){

            matrizLEDs[i][1] = 'X';
            matrizLEDs[i][4] = 'X';
            matrizLEDs[i][7] = 'X';
            matrizLEDs[j][0] = 'X';
            matrizLEDs[j][3] = 'X';
            matrizLEDs[j][6] = 'X';
            matrizLEDs[j+1][0] = 'X';
            matrizLEDs[j+1][3] = 'X';
            matrizLEDs[j+1][6] = 'X';
            matrizLEDs[j+2][2] = 'X';
            matrizLEDs[j+2][5] = 'X';
            matrizLEDs[j+3][2] = 'X';
            matrizLEDs[j+3][5] = 'X';
            matrizLEDs[j+4][0] = 'X';
            matrizLEDs[j+4][3] = 'X';
            matrizLEDs[j+4][6] = 'X';
            matrizLEDs[j+5][0] = 'X';
            matrizLEDs[j+5][3] = 'X';
            matrizLEDs[j+5][6] = 'X';
            matrizLEDs[j+6][2] = 'X';
            matrizLEDs[j+6][5] = 'X';
            matrizLEDs[j+7][2] = 'X';
            matrizLEDs[j+7][5] = 'X';
        }

    }

    for (int i = 0; i < 8; i++) {
        for(int j=0; i<8; i++){

            matrizLEDs[i][3] = 'X';
            matrizLEDs[i][4] = 'X';
            matrizLEDs[i/4 + 3][0] = 'X';
            matrizLEDs[i/4 + 3][7] = 'X';
            matrizLEDs[i/2 + 2][1] = 'X';
            matrizLEDs[i/2 + 2][6] = 'X';
            matrizLEDs[i/2 + 2][2] = 'X';
            matrizLEDs[i/2 + 2][5] = 'X';
            matrizLEDs[j+1][2] = 'X';
            matrizLEDs[j+6][2] = 'X';
            matrizLEDs[j+1][5] = 'X';
            matrizLEDs[j+6][5] = 'X';

        }
    }


    for (int i = 0; i < 8; i++) {
        for(int j=0; i<8; i++){

            matrizLEDs[i][3] = 'X';
            matrizLEDs[i/4 + 3][6] = 'X';
            matrizLEDs[i/2 + 2][5] = 'X';
            matrizLEDs[i/2 + 2][4] = 'X';
            matrizLEDs[j + 1][4] = 'X';
            matrizLEDs[j + 6][4] = 'X';
            matrizLEDs[j][0] = 'X';
            matrizLEDs[j + 7][0] = 'X';
            matrizLEDs[i/4][1] = 'X';
            matrizLEDs[i/4 + 6][1] = 'X';
            matrizLEDs[i/3][2] = 'X';
            matrizLEDs[i/3 + 5][2] = 'X';

        }
    }

    // Mostrar la matriz
    for (int i = 0; i < 8; i++) {
        for (int j = 0; j < 8; j++) {
            cout << matrizLEDs[i][j] << ' ';
        }
        cout << endl;
    }
}

int main() {
    mostrarPatronX();
    return 0;
}
